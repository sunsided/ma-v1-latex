	]

	\addplot[
			id=GaussAt1Sigma,	
			black,
			mark=*,
			samples=10,
			domain=1E-2:1E2] 
		gnuplot{
			  1/(x*sqrt(2*pi))*exp(-(1*x)**2/(2*x**2)) 
			};
	\addlegendentry{$x=1\sigma$};

	\addplot[
			id=GaussAt3Sigma,	
			blue,
			mark=square,
			samples=10,

	\addplot[
			id=GaussAt6Sigma,	
			red,
			mark=triangle,
			samples=10,
			domain=1E-2:1E2] 
		gnuplot{
			  1/(x*sqrt(2*pi))*exp(-(6*x)**2/(2*x**2))

	\addlegendentry{$x=6\sigma$};	
			
	\addplot[
			id=GaussAt9Sigma,

		gnuplot{
			  1/(x*sqrt(2*pi))*exp(-(9*x)**2/(2*x**2)) 
			};
	\addlegendentry{$x=9\sigma$};	
	  
	
	\end{loglogaxis}
